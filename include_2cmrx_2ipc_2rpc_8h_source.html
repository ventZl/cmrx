62/*
63 * Checker to make sure that the layout of type being used to perform calls is correct.
64 * The CMRX ABI requires, that the `vtable` member is the first member in the struct.
65 * Otherwise kernel won't be able to locate the virtual method table during rpc_call()
66 * execution.
67 */

73/*
74 * The master RPC call macro.
75 * This macro will emit several checks:
76 * - check that RPC service instance is a pointer to something.
77 * - check that va_args are actually compatible to what RPC method expects.
78 * - check the layout of RPC service (especially the position of the VTable)
79 * If all the checks will pass then RPC call is emitted. Note that all
80 * the checking is performed in the compile time.
81 */
82
83#define CMRX_RPC_CALL(service_instance, method_name, ...) \
84 CMRX_RPC_EVALUATOR(CMRX_RPC_GET_ARG_COUNT(__VA_ARGS__))(\
85 (service_instance), \
86 offsetof(typeof(*((service_instance)->vtable)), method_name) / sizeof(void *), \
87 ##__VA_ARGS__); \
88 CMRX_RPC_SERVICE_FORM_CHECKER(service_instance); \
89 CMRX_RPC_TYPE_CHECKER(CMRX_RPC_GET_ARG_COUNT(__VA_ARGS__), (service_instance)->vtable->method_name, __VA_ARGS__) \
90 CMRX_RPC_INTERFACE_CHECKER(service_instance); \
