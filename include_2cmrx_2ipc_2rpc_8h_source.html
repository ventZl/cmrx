70/*
71 * Checker to make sure that the layout of type being used to perform calls is correct.
72 * The CMRX ABI requires, that the `vtable` member is the first member in the struct.
73 * Otherwise kernel won't be able to locate the virtual method table during rpc_call()
74 * execution.
75 */
76
77#define CMRX_RPC_INTERFACE_CHECKER_IMPL(service_instance) "Service `" #service_instance "` has invalid layout! VTable must be the first member!"
78#define CMRX_RPC_INTERFACE_CHECKER_AUX(service_instance) CMRX_RPC_INTERFACE_CHECKER_IMPL(service_instance)
79#define CMRX_RPC_INTERFACE_CHECKER(service_instance) _Static_assert(offsetof(typeof(*(service_instance)), vtable) == 0, CMRX_RPC_INTERFACE_CHECKER_AUX(service_instance))

81/*
82 * The master RPC call macro.
83 * This macro will emit several checks:
84 * - check that RPC service instance is a pointer to something.
85 * - check that va_args are actually compatible to what RPC method expects.
86 * - check the layout of RPC service (especially the position of the VTable)
87 * If all the checks will pass then RPC call is emitted. Note that all
88 * the checking is performed in the compile time.
89 */
90
91#define CMRX_RPC_CALL(service_instance, method_name, ...) \
92 CMRX_RPC_EVALUATOR(CMRX_RPC_GET_ARG_COUNT(__VA_ARGS__))(\
93 (service_instance), \
94 offsetof(typeof(*((service_instance)->vtable)), method_name) / sizeof(void *), \
95 ##__VA_ARGS__); \
96 CMRX_RPC_SERVICE_FORM_CHECKER(service_instance); \
97 CMRX_RPC_TYPE_CHECKER(CMRX_RPC_GET_ARG_COUNT(__VA_ARGS__), (service_instance)->vtable->method_name, ##__VA_ARGS__) \
98 CMRX_RPC_INTERFACE_CHECKER(service_instance); \
