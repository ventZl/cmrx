195static inline ExceptionFrame * push_exception_frame(ExceptionFrame * frame, unsigned args)
196{
197 ExceptionFrame * outframe = (ExceptionFrame *) (((uint32_t *) frame) - (EXCEPTION_FRAME_SIZE + args));
198 bool padding = false;
199
200 // Check if forged frame is 8-byte aligned, or not
201 if ((((uint32_t) outframe) % 8) != 0)
202 {
203 // Frame needs padding
204 outframe = (ExceptionFrame *) (((uint32_t *) outframe) - 1);
205 padding = true;
206 }
207
208 outframe->xpsr = frame->xpsr;
209
210 if (padding)
211 {
212 // we have padded the stack frame, clear STKALIGN in order to let
213 // CPU know that original SP was 4-byte aligned
214 outframe->xpsr |= 1 << 9;
215 }
216 else
217 {
218 // we didn't pad the stack frame, set STKALIGN in order to let
219 // CPU know that original SP was 8-byte aligned
220 outframe->xpsr &= ~(1 << 9);
221 }
222
223 return outframe;
224}

235static inline ExceptionFrame * shim_exception_frame(ExceptionFrame * frame, unsigned args)
236{
237 ExceptionFrame * outframe = (ExceptionFrame *) (((uint32_t *) frame) - args);
238 bool padding = false;
239
240 // Check if forged frame is 8-byte aligned, or not
241 if ((((uint32_t) outframe) % 8) != 0)
242 {
243 // Frame needs padding
244 outframe = (ExceptionFrame *) (((uint32_t *) outframe) - 1);
245 padding = true;
246 }
247
248 /* This has to be done from lowest address to highest to avoid
249 * overwriting usable data.
250 */
251
252 for (unsigned int q = 0; q < sizeof(ExceptionFrame) / 4; ++q)
253 {
254 ((uint32_t*) outframe)[q] = ((uint32_t*) frame)[q];
255 }
256 if (padding)
257 {
258 // we have padded the stack frame, clear STKALIGN in order to let
259 // CPU know that original SP was 4-byte aligned
260 outframe->xpsr |= 1 << 9;
261 }
262 else
263 {
264 // we didn't pad the stack frame, set STKALIGN in order to let
265 // CPU know that original SP was 8-byte aligned
266 outframe->xpsr &= ~(1 << 9);
267 }
268
269 return outframe;
270}

280static inline ExceptionFrame * pop_exception_frame(ExceptionFrame * frame, unsigned args)
281{
282 ExceptionFrame * outframe = (ExceptionFrame *) (((uint32_t *) frame) + (EXCEPTION_FRAME_SIZE + args));
283 if (((frame->xpsr >> 9) & 1) == 1)
284 {
285 outframe = (ExceptionFrame *) (((uint32_t *) outframe) + 1);
286 }
287
288 // rewrite xPSR from pop-ped frame, retain value of bit 9
289 outframe->xpsr = (outframe->xpsr & (1 << 9)) | (frame->xpsr & ~(1 << 9));
290
291 return outframe;
292}
